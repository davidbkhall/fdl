29inline int64_t bankers_round(double value) {
30 double rounded = std::round(value);
31 // Check if we're exactly at the halfway point
32 double const remainder = value - std::floor(value);
33 if (std::abs(remainder - constants::kHalfway) < constants::kFpHalfwayTolerance) {
34 // Halfway case: round to even
35 auto r = static_cast<int64_t>(rounded);
36 if (r % constants::kEvenDivisor != 0) {
37 // rounded is odd, go the other way
38 rounded = std::floor(value + constants::kHalfway);
39 if (static_cast<int64_t>(rounded) % constants::kEvenDivisor != 0) {
40 rounded = std::ceil(value - constants::kHalfway);
41 }
42 }
43 }
44 return static_cast<int64_t>(rounded);
